128int compile_and_execute(const char* code, char* output, size_t output_size) {
129 char log_msg[256];
130
131 // Write code to temporary file
132 FILE* temp_file = fopen("temp_code.c", "w");
133 if (!temp_file) {
134 snprintf(output, output_size, "ERROR: Cannot create temporary file\n");
135 return -1;
136 }
137 fprintf(temp_file, "%s", code);
138 fclose(temp_file);
139
140 pthread_mutex_lock(&stats_mutex);
141 total_compilations++;
142 pthread_mutex_unlock(&stats_mutex);
143
144 // Compile the code
145 int compile_result = system("gcc temp_code.c -o temp_program 2> compile_error.log");
146
147 if (compile_result != 0) {
148 // Compilation failed
149 FILE* error_file = fopen("compile_error.log", "r");
150 if (error_file) {
151 fread(output, 1, output_size - 1, error_file);
152 fclose(error_file);
153 } else {
154 snprintf(output, output_size, "ERROR: Compilation failed\n");
155 }
156 snprintf(log_msg, sizeof(log_msg), "Compilation failed");
157 log_activity(log_msg);
158 return -1;
159 }
160
161 // Execute the program
162 FILE* exec_pipe = popen("timeout 5 ./temp_program 2>&1", "r");
163 if (!exec_pipe) {
164 snprintf(output, output_size, "ERROR: Cannot execute program\n");
165 return -1;
166 }
167
168 size_t bytes_read = fread(output, 1, output_size - 1, exec_pipe);
169 output[bytes_read] = '\0';
170
171 int exec_result = pclose(exec_pipe);
172
173 pthread_mutex_lock(&stats_mutex);
174 if (exec_result == 0) {
175 successful_compilations++;
176 }
177 pthread_mutex_unlock(&stats_mutex);
178
179 // Clean up
180 unlink("temp_code.c");
181 unlink("temp_program");
182 unlink("compile_error.log");
183
184 snprintf(log_msg, sizeof(log_msg), "Code executed, result: %d", exec_result);
185 log_activity(log_msg);
186
187 return exec_result;
188}

209void* handle_client(void* arg) {
210 client_info_t* client = (client_info_t*)arg;
211 char buffer[BUFFER_SIZE];
212 char output[BUFFER_SIZE];
213
214 log_activity("Regular client connected");
215
216 while (1) {
217 memset(buffer, 0, BUFFER_SIZE);
218 int bytes_received = recv(client->socket, buffer, BUFFER_SIZE - 1, 0);
219
220 if (bytes_received <= 0) {
221 break;
222 }
223
224 if (strncmp(buffer, "QUIT", 4) == 0) {
225 break;
226 }
227
228 // Compile and execute the received code
229 compile_and_execute(buffer, output, sizeof(output));
230
231 // Send result back to client
232 send(client->socket, output, strlen(output), 0);
233 }
234
235 log_activity("Regular client disconnected");
236 close(client->socket);
237 free(client);
238 return NULL;
239}

274
275 if (strncmp(buffer, "STATUS", 6) == 0) {
276 pthread_mutex_lock(&stats_mutex);

279 total_compilations, successful_compilations,
280 total_compilations - successful_compilations);
281 pthread_mutex_unlock(&stats_mutex);
282 } else if (strncmp(buffer, "SHUTDOWN", 8) == 0) {
283 snprintf(response, sizeof(response), "Server shutting down...\n");
284 send(client->socket, response, strlen(response), 0);
285 server_running = 0;
286 break;
287 } else if (strncmp(buffer, "LOGS", 4) == 0) {
288 FILE* log_file = fopen("server.log", "r");
289 if (log_file) {
290 size_t bytes_read = fread(response, 1, sizeof(response) - 1, log_file);
291 response[bytes_read] = '\0';
292 fclose(log_file);
293 } else {
294 snprintf(response, sizeof(response), "No logs available\n");
295 }
296 } else if (strncmp(buffer, "QUIT", 4) == 0) {
297 break;
298 } else {
299 snprintf(response, sizeof(response), "Unknown command. Available: STATUS, LOGS, SHUTDOWN, QUIT\n");
300 }
301
302 send(client->socket, response, strlen(response), 0);
303 }
304
305 log_activity("Admin client disconnected");

330void* regular_server_thread(void* arg) {
331 int server_fd, client_socket;
332 struct sockaddr_in address;
333 int opt = 1;
334 int addrlen = sizeof(address);
335
336 if ((server_fd = socket(AF_INET, SOCK_STREAM, 0)) == 0) {
337 perror("socket failed");
338 exit(EXIT_FAILURE);
339 }
340
341 if (setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR | SO_REUSEPORT, &opt, sizeof(opt))) {
342 perror("setsockopt");
343 exit(EXIT_FAILURE);
344 }
345
346 address.sin_family = AF_INET;
347 address.sin_addr.s_addr = INADDR_ANY;
348 address.sin_port = htons(PORT);
349
350 if (bind(server_fd, (struct sockaddr *)&address, sizeof(address)) < 0) {
351 perror("bind failed");
352 exit(EXIT_FAILURE);
353 }
354
355 if (listen(server_fd, MAX_CLIENTS) < 0) {
356 perror("listen");
357 exit(EXIT_FAILURE);
358 }
359
360 printf("Regular client server listening on port %d\n", PORT);
361 log_activity("Regular client server started");
362
363 while (server_running) {
364 if ((client_socket = accept(server_fd, (struct sockaddr *)&address, (socklen_t*)&addrlen)) < 0) {
365 if (server_running) {
366 perror("accept");
367 }
368 continue;
369 }
370
371 client_info_t* client = malloc(sizeof(client_info_t));
372 client->socket = client_socket;
373 strcpy(client->type, "regular");
374
375 pthread_t thread;
376 pthread_create(&thread, NULL, handle_client, client);
377 pthread_detach(thread);
378 }
379
380 close(server_fd);
381 return NULL;
382}
